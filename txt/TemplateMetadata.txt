import type { TemplateMetadata } from "../../types/TemplateMetadata";

interface TemplateMetadataBarProps {
  metadata: TemplateMetadata;
  onChange: (updated: Partial<TemplateMetadata>) => void;
  onSave: () => void;
  isSaving: boolean;
  saveSuccess: boolean;
  saveError: string | null;

//   onChange: (updater: (prev: any) => any) => void;
//   onSave: () => void;
  onValidate: () => void;
  onPreview: () => void;

}




export function TemplateMetadataBar({
  metadata,
  onChange,
  onSave,
  onValidate,
  onPreview,
  isSaving,
  saveError,
  saveSuccess
}: TemplateMetadataBarProps) {

  if (!metadata) {
    return <div style={{ padding: 12 }}>Loadingâ€¦</div>;
  }

// console.log("TemplateMetadataBar useParams:", useParams());

  return (
    <div
      style={{
        display: "flex",
        alignItems: "center",
        gap: 16,
        padding: 12,
        borderBottom: "1px solid #ddd",
        background: "#fafafa",
      }}
    >
<input
  className="form-control"
  value={metadata.name}
onChange={(e) =>
  onChange({
    name: e.target.value
  })
}

  style={{ fontSize: 18 }}
/>


      {/* Save button */}
      <button
        type="button"
        className="btn btn-primary"
        onClick={() => {
          console.log("Save button clicked");
          onSave();
        }}
        disabled={isSaving}
      >
        {isSaving ? "Saving..." : "Save"}
      </button>

      {/* Save status */}
      {saveSuccess && (
        <span style={{ color: "green" }}>
          Saved successfully
        </span>
      )}

      {saveError && (
        <span style={{ color: "red" }}>
          {saveError}
        </span>
      )}

      {/* Other actions */}
      {saveSuccess && (
        <span style={{ color: "green" }}>
          Saved successfully
        </span>
      )}

      {saveError && (
        <span style={{ color: "red" }}>
          {saveError}
        </span>
      )}

      <button onClick={onValidate} className="btn btn-primary">Validate</button>
      <button onClick={onPreview} className="btn btn-primary">Preview</button>
    </div>
  );
}

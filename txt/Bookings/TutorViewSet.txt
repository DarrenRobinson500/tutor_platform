    @action(detail=True, methods=["POST"])
    def confirm_booking(self, request, pk=None):
        tutor = self.get_object()
        booking_id = request.data.get("id")
        booking_type = request.data.get("type")

        if not booking_id or not booking_type:
            return Response({"ok": False, "error": "Missing id or type."})

        booking_model = BookingWeekly
        if booking_type == "adhoc": booking_model = BookingAdhoc
        try:
            b = booking_model.objects.get(id=booking_id, tutor=tutor)
            b.confirmed = not b.confirmed
            b.save()
            update_booking_confirmed_in_cache(tutor.id, booking_id, booking_type, b.confirmed)
            return Response({"ok": True, "confirmed": b.confirmed})
        except booking_model.DoesNotExist:
            return Response({"ok": False, "error": "Booking not found."})
